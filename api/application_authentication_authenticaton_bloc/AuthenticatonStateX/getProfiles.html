List<InfluencerProfileDetails> get getProfiles {
  final list = <InfluencerProfileDetails>[];
  for (var i in sortedSelectedPlatforms) {
    final link = profileLinks[i];
    switch (link.urlType) {
      case PostLinkType.instagram:
        list.add(
          InfluencerProfileDetails(
            profileType: link.urlType?.toName,
            link: link.getValue,
            followers: instaFollowerCount,
            following: instaFollowingCount,
            postCount: postCount,
            isApproved: link.isChanged ? 0 : 1,
          ),
        );

      case PostLinkType.facebook:
        list.add(
          InfluencerProfileDetails(
            profileType: link.urlType?.toName,
            link: link.getValue,
            followers: facebookFollowerCount,
            following: facebookFollowingCount,
            isApproved: link.isChanged ? 0 : 1,
          ),
        );
      case PostLinkType.tiktok:
        list.add(
          InfluencerProfileDetails(
            profileType: link.urlType?.toName,
            link: link.getValue,
            followers: tiktokFollowerCount,
            following: tiktokFollowingCount,
            likeCount: likeCount,
            isApproved: link.isChanged ? 0 : 1,
          ),
        );
      case PostLinkType.youtube:
        list.add(
          InfluencerProfileDetails(
            profileType: link.urlType?.toName,
            link: link.getValue,
            subscriberCount: subscriberCount,
            videoCount: videoCount,
            isApproved: link.isChanged ? 0 : 1,
          ),
        );
      default:
    }
  }

  return list;
}
