Future<void> getLocation(Position position, {int? userId}) async {
  if (state.placeMark != null) {
    updateLocation(
      UpdatePositionParam(
        userId: userId,
        latitude: position.latitude,
        longitude: position.longitude,
        country: state.placeMark!.country,
        state: state.placeMark!.state,
        city: state.placeMark!.localArea ?? state.placeMark!.district,
      ),
    );
    return;
  }
  final result = await getUserCurrentUserPosition.getLocationFromCordinates(
    latitude: '${position.latitude}',
    longitude: '${position.longitude}',
  );

  result.fold(
    (l) {},
    (location) {
      emit(
        state.copyWith(
          placeMark: location,
        ),
      );
      updateLocation(
        UpdatePositionParam(
          userId: userId,
          latitude: position.latitude,
          longitude: position.longitude,
          country: location.country,
          state: location.state,
          city: location.localArea ?? location.district,
        ),
      );
    },
  );
}
