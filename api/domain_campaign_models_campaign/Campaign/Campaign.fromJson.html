factory Campaign.fromJson(Map<String, dynamic> json) {
  return Campaign(
    broadcastLinks: json['broadcast_links'] != null
        ? List<String>.from(json['broadcast_links'])
        : [],
    currencySymbol: json['currency_symbol'],
    countryCode: json['country_code'],
    commentQurstion: json['comment_question'],
    id: json["id"],
    campaignId: json["campaign_id"],
    campaignCode: json["campaign_code"],
    clientUserId: json["client_user_id"],
    postLink: json["post_link"],
    conductedAt:
        DateTime.tryParse(json['campaign_preference']["updated_at"] ?? ""),
    platform: json["platform"],
    giveawayValue: json["giveaway_value"],
    giveawayName: json["giveaway_name"],
    language: json["language"],
    campaignStartDate: DateTime.tryParse(
      json["campaign_start_date"] ?? json["campaign_start"] ?? '',
    ),
    campaignStart: DateTime.tryParse(json["campaign_start"] ?? ""),
    campaignEnd: DateTime.tryParse(json["campaign_end"] ?? ''),
    campaignStatus: json["campaign_status"],
    campaignReactionCount: json["campaign_reaction_count"],
    campaignViewCount: json["campaign_view_count"],
    involvementType: json["involvement_type"],
    boostId: json["boost_id"],
    boostedOn: json["boosted_on"] != null
        ? DateTime.tryParse(json["boosted_on"]!)
        : null,
    isBoosted: json["is_boosted"],
    campaignPreference: json["campaign_preference"] == null
        ? null
        : CampaignPreference.fromJson(json["campaign_preference"]),
    clientUser: json["client_user"] == null
        ? null
        : UserDetails.fromJson(json["client_user"]),
  );
}
