factory WinnerDetails.fromJson(Map<String, dynamic> json) {
  return WinnerDetails(
    id: json["id"],
    username: json["username"],
    profileUrl: json["profile_url"],
    imageUrl: json["image_url"],
    mentions: json["mentions"] != null
        ? json["mentions"] is String
            ? List<String>.from(jsonDecode(json["mentions"]))
            : json["mentions"]
        : [],
    comments: json["comments"],
    winnerType: json["winner_type"],
    position: json["position"],
    giveawayDetailId:
        json["giveaway_detail_id"] ?? json["campaign_detail_id"],
    platform: json["platform"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    commentedAt: DateTime.tryParse(json["commented_at"] ?? ""),
  );
}
