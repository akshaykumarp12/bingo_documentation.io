@override
ResultFuture<String> saveWinners({
  required Params params,
  required GiveawayFormat format,
}) async {
  final config = _saveWinnersConfig[format];

  if (config == null) {
    return left(
      const MainFailure.customError(
        errorMsg: 'Unknown Giveaway Format',
      ),
    );
  }

  saveWinnerContext
    ..setStrategy = config['strategy']
    ..setUrl = config['url'];

  return saveWinnerContext.saveWinners(params);
}
