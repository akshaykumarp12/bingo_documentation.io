static Widget build({
  required PostLinkType type,
  required bool repetedTags,
}) {
  PostLayoutBuilder builder;
  switch (type) {
    case PostLinkType.unknown:
    case PostLinkType.tiktok:
      builder = TiktokDrawLayoutBuilder();
    case PostLinkType.youtube:
      builder = YoutubeDrawLayoutBuilder();
    case PostLinkType.instagram:
      builder = InstagramDrawLayoutBuilder();
    case PostLinkType.facebook:
      builder = FacebookWidgetBuilder();
  }

  return builder.build(repetedTags: repetedTags);
}
