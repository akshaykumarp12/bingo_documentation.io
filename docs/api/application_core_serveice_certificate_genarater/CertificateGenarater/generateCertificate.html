static ResultFuture<File> generateCertificate({
  required List<Winner> winners,
  required String giveAwayCode,
  required String giveAwayName,
  required String noOfParticipants,
}) async {
  String formattedDate =
      DateFormat('dd-MM-yyyy HH:mm:ss').format(DateTime.now());

  final pdf = pw.Document();
  final certificate = await rootBundle.load(AppAssets.certificateFrameImage);
  final certificateImage = certificate.buffer.asUint8List();

  List<pw.MemoryImage> items = [];

  for (var winner in winners.take(4)) {
    Uint8List? imageByte;
    int retryCount = 3;

    while (imageByte == null && retryCount > 0) {
      imageByte = await fetchImage(
        retryCount <= 1 ? profileImage : winner.imageUrl,
      );
      retryCount--;
    }

    if (imageByte != null) {
      items.add(pw.MemoryImage(imageByte));
    }
  }

  pdf.addPage(
    pw.Page(
      margin: pw.EdgeInsets.zero,
      pageFormat: PdfPageFormat.a4,
      build: (pw.Context context) {
        return pw.Stack(
          alignment: pw.Alignment.bottomCenter,
          children: [
            pw.Container(
              height: double.infinity,
              width: double.infinity,
              padding:
                  const pw.EdgeInsets.only(top: 275, left: 50, right: 50),
              decoration: pw.BoxDecoration(
                image: pw.DecorationImage(
                  image: pw.MemoryImage(certificateImage),
                  fit: pw.BoxFit.fill,
                ),
              ),
              child: pw.Column(
                mainAxisAlignment: pw.MainAxisAlignment.start,
                children: [
                  //  pw.SizedBox(height: ),
                  // pw.SvgImage(svg: logo, width: 60, height: 60),
                  // pw.SizedBox(height: 20),
                  // pw.Text('CERTIFICATE\nOF WINNER',
                  //     textAlign: pw.TextAlign.center,
                  //     style: pw.TextStyle(
                  //       fontSize: 22,
                  //       fontWeight: pw.FontWeight.bold,
                  //     )),
                  // pw.SizedBox(height: 20),
                  pw.Wrap(
                    spacing: 8,
                    runSpacing: 8,
                    children: List.generate(
                        winners.length >= 4 ? 4 : winners.length, (index) {
                      return pw.Container(
                          padding:
                              const pw.EdgeInsets.symmetric(horizontal: 8),
                          width: 100,
                          child: pw.Column(
                            children: [
                              pw.Container(
                                width: 80,
                                height: 80,
                                child: pw.ClipOval(
                                  child: pw.Image(
                                    items[index],
                                    fit: pw.BoxFit.cover,
                                  ),
                                ),
                              ),
                              pw.SizedBox(
                                width: 100,
                                child: pw.Text(
                                  winners[index].username ?? '',
                                  style: pw.TextStyle(
                                    fontSize: 18,
                                    fontWeight: pw.FontWeight.bold,
                                  ),
                                  maxLines: 1,
                                  overflow: pw.TextOverflow.clip,
                                ),
                              )
                            ],
                          ));
                    }),
                  ),
                  if (winners.length > 4)
                    pw.Column(
                      children: [
                        pw.SizedBox(height: 20),
                        pw.Text('More Winners',
                            style: pw.TextStyle(
                                fontSize: 22,
                                fontWeight: pw.FontWeight.bold)),
                        pw.SizedBox(height: 20),
                        pw.Wrap(
                          spacing: 12,
                          runSpacing: 12,
                          children: List.generate(
                            winners
                                .sublist(4,
                                    winners.length > 13 ? 13 : winners.length)
                                .length,
                            (index) {
                              return pw.Container(
                                padding: const pw.EdgeInsets.symmetric(
                                  horizontal: 16,
                                ),
                                width: 120,
                                height: 40,
                                decoration: pw.BoxDecoration(
                                  border: pw.Border.all(
                                    color: PdfColors.black,
                                    width: 2,
                                    style: pw.BorderStyle
                                        .dashed, // Creates a dotted border
                                  ),
                                ),
                                alignment: pw.Alignment.center,
                                child: pw.SizedBox(
                                  width: 150,
                                  child: pw.Text(
                                    winners.sublist(4)[index].username ?? '',
                                    style: pw.TextStyle(
                                      fontSize: 18,
                                      fontWeight: pw.FontWeight.bold,
                                    ),
                                    maxLines: 1,
                                    overflow: pw.TextOverflow.clip,
                                  ),
                                ),
                              );
                            },
                          ),
                        ),
                      ],
                    ),
                  pw.SizedBox(height: 20),
                  pw.Text(giveAwayName.capitalize,
                      style: pw.TextStyle(
                          fontSize: 22, fontWeight: pw.FontWeight.bold)),
                  pw.SizedBox(height: 10),
                  pw.Text('Selected from $noOfParticipants Participants',
                      style: pw.TextStyle(
                          fontSize: 22, fontWeight: pw.FontWeight.normal)),
                  pw.SizedBox(height: 20),
                  pw.Container(
                    width: 350,
                    height: 80,
                    padding: const pw.EdgeInsets.all(10),
                    decoration: pw.BoxDecoration(
                      border: pw.Border.all(
                        color: PdfColors.black,
                        width: 2,
                        style:
                            pw.BorderStyle.dashed, // Creates a dotted border
                      ),
                    ),
                    alignment: pw.Alignment.center,
                    child: pw.Text(
                      giveAwayCode,
                      style: pw.TextStyle(
                        fontSize: 20,
                        fontWeight: pw.FontWeight.bold,
                      ),
                    ),
                  ),
                ],
              ),
            ),
            pw.Positioned(
              bottom: 50,
              child: pw.Row(
                mainAxisAlignment: pw.MainAxisAlignment.center,
                children: [
                  pw.RichText(
                    textAlign: pw.TextAlign.center,
                    text: pw.TextSpan(
                      style: pw.TextStyle(
                        color: PdfColor.fromHex('#59516B'),
                        fontSize: 10,
                        fontWeight: pw.FontWeight.normal,
                      ),
                      children: [
                        const pw.TextSpan(text: "Note: "),
                        pw.TextSpan(
                          text:
                              "This is an authentic Bingo certificate. You can verify its authenticity by entering the\nCertificate Code in the Bingo app or on www.bingogiveaway.com",
                          style: pw.TextStyle(
                            color: PdfColor.fromHex('#59516B'),
                            fontSize: 8,
                            fontWeight: pw.FontWeight.normal,
                          ),
                        )
                      ],
                    ),
                  ),
                ],
              ),
            ),
            pw.Positioned(
              bottom: 17,
              right: 175,
              child: pw.Text(
                formattedDate,
                style: pw.TextStyle(
                  color: PdfColor.fromHex('#59516B'),
                  fontSize: 6,
                  fontWeight: pw.FontWeight.normal,
                ),
              ),
            ),
          ],
        );
      },
    ),
  );

  try {
    final data = await pdf.save();

    final path = await savePdf(data);
    return right(path);
  } catch (e) {
    // Logger.logSuccess(e);
    return left(
      const MainFailure.genericError(errorMsg: 'Failed To Genarate PDF'),
    );
  }
}
