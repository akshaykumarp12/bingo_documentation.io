PickWinnerBloc({
  required this.winnersCubit,
  required this.giveawayDetailsBloc,
  required this.iPickWinnerFacade,
  required this.postDetailsBloc,
  required this.ratingCubit,
  required this.homeUiCubit,
  required this.authenticatonBloc,
}) : super(PickWinnerState.initial()) {
  on<WinnerPicked>(_winnerPicked);
  on<FirstWinnerPicked>(_firstWinnerPicked);
  on<StartRecordingVideo>(_startRecordingVideo);
  on<StopRecordingVideo>(
    _stopRecordingVideo,
    transformer: debounce(
      const Duration(seconds: 2),
    ),
  );
  on<ClearPickWinnerState>(_clearPickWinnerState);
  on<LoadWaterMark>(_loadWaterMark);
  on<ChangeOpacity>(_changeOpacity);
  on<PauseRecording>(_pauseRecording);
  on<ResumeRecording>(_resumeRecording);
  on<RecordVideo>(_recordVideo);
  on<PreloadConfetti>(_preloadConfetti);
  on<SaveGiveaway>(
    _saveGiveaway,
    transformer: droppable(),
  );
  on<GetComments>(
    _getComments,
    transformer: droppable(),
  );
  on<AddComments>(_addComments);
  on<PickBackupWinners>(_pickBackupWinners);
  on<SaveWinners>(
    _saveWinners,
    transformer: droppable(),
  );
  on<FilterComments>(_filterComments);
  on<GiveawayFormatChanged>(_giveawayFormatChanged);
  on<CommentFetchStatusChanged>(_commentFetchStatusChanged);
  on<_DeleteGiveaway>(_deleteGiveaway);
  on<UpdateStarRateing>(_updateStarRateing);
  on<AddProgress>(_addProgress, transformer: sequential());
  on<UploadWinnerCertificate>(_uploadWinnerCertificate);
  on<MessageChanged>(_messageChanged);
  on<GenerateCertificate>(_generateCertificate);

  on<RemovePickedWinnerFromList>(_removePickedWinnerFromList);

  // STREAM THAT LISTEN FOR PICKED WinnersState

  winnerStateListener = winnersCubit.stream.listen((winnerState) {
    // Logger.logSuccess('winnersCubit');
    winnersState = winnerState;
  });
  // STREAM THAT LISTEN FOR GiveawayDetailsState
  giveawayStateListener =
      giveawayDetailsBloc.stream.listen((giveawayDetails) {
    final d = {
      'totalWinners': giveawayDetails.totalWinners,
      'backupWinners': giveawayDetails.backupWinners,
      'totalMentions': giveawayDetails.totalMentions,
      'totalCommentLikes': giveawayDetails.totalCommentLikes,
      'commentReplay': giveawayDetails.commentReplay,
      'keyWord': giveawayDetails.keyWord,
      'unwantedWord': giveawayDetails.unwantedWord,
      'start': giveawayDetails.start,
      'end': giveawayDetails.end,
    };
    Logger.logWarning('giveawayDetails $d');
    giveawayDetailsState = giveawayDetails;
  });

  postStateListener = postDetailsBloc.stream.listen((postDetails) {
    postDetailsState = postDetails;
  });
}
