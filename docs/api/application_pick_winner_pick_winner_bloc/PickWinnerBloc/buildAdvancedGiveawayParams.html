Params? buildAdvancedGiveawayParams() {
  final type = giveawayDetailsState?.giveawayType;
  if (type == null) return null;

  return type == GiveawayType.advanced
      ? ParamBuilder()
          .addParam(
              key: 'mention_count',
              value: giveawayDetailsState?.totalMentions)
          .addParam(
              key: 'hashtag_count',
              value: giveawayDetailsState?.totalHashtags)
          .addParam(
              key: 'share_count', value: giveawayDetailsState?.shareCount)
          .addParam(
              key: 'like_count',
              value: giveawayDetailsState?.totalCommentLikes)
          .addParam(
              key: 'tags_repetition',
              value: giveawayDetailsState?.noRepetedTags)
          .addParam(
              key: 'comment_likes_count',
              value: giveawayDetailsState?.totalCommentLikes)
          .addParam(
              key: 'comment_reply_count',
              value: giveawayDetailsState?.commentReplay)
          .addParam(
              key: 'location', value: giveawayDetailsState?.selectedCountries)
          .addParam(
              key: 'date_from',
              value: giveawayDetailsState?.start.getValue?.ymd)
          .addParam(
              key: 'date_to', value: giveawayDetailsState?.end.getValue?.ymd)
          .addParam(
              key: 'follow_pages', value: giveawayDetailsState?.followOptions)
          .addParam(key: 'keywords', value: giveawayDetailsState?.keyWord)
          .addParam(
              key: 'unwanted_words',
              value: giveawayDetailsState?.unwantedWord)
          .addParam(
              key: 'post_count', value: giveawayDetailsState?.minimumNoOfPost)
          .addParam(
              key: 'follower_count',
              value: giveawayDetailsState?.minimumNoOfFollowers)
          .addParam(
              key: 'profile_picture_required',
              value: giveawayDetailsState?.profileRequired)
          .addParam(
              key: 'name_required', value: giveawayDetailsState?.nameRequired)
          .addParam(
              key: 'biography_required',
              value: giveawayDetailsState?.biograpyRequired)
          .addParam(
              key: 'comment_likes',
              value: giveawayDetailsState?.totalCommentLikes)
          .build()
      : null;
}
