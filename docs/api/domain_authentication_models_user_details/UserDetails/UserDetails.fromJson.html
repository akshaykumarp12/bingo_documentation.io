factory UserDetails.fromJson(Map<String, dynamic> json) {
  return UserDetails(
    upgradeToInfluencer: json['upgrade_to_influencer'] == 1,
    photoUrl: json['photo_url'],
    id: json["id"],
    userId: json["user_id"],
    deviceId: json["device_id"],
    socialId: json["social_id"],
    socialName: json["social_name"] ?? json["username"],
    photo: json["photo"],
    username: json["username"],
    email: json["email"],
    password: json["password"],
    fcmToken: json["fcm_token"],
    score: json["score"],
    profileLinks: (json["profile_links"] as List? ?? [])
        .map((data) => InfluencerProfileDetails.fromJson(data))
        .toList(),
    loginWith: json["login_with"],
    influencer: json["influencer"],
    potentialStatus: json["potential_status"],
    coin: json["coin"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    deletedAt: json["deleted_at"],
    apiToken: json["api_token"],
    userType: json['user_type'],
  );
}
