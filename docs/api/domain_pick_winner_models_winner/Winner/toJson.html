Map<String, dynamic> toJson(GiveawayFormat format) {
  var data = {
    "username": username,
    "profile_url": profileUrl,
    "platform": platform,
    "winner_type": winnerType,
    "position": position,
    "mentions": mentions.map((x) => x).toList(),
    "comments": comments,
    'image_url': imageUrl,
    'comment_likes': commentLikes,
    'comment_replies': commentReplies,
    'instagram_owner_id': instagramOwnerId,
    'is_private': isPrivate,
    'commentor_timestamp': commentedAt?.ymdHMS,
  };

  switch (format) {
    case GiveawayFormat.giveaway:
      data['giveaway_detail_id'] = id;
      break;
    case GiveawayFormat.campaign:
      data['campaign_detail_id'] = id;
  }

  return data;
}
