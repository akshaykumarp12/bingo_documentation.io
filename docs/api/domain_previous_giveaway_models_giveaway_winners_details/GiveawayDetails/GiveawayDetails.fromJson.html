factory GiveawayDetails.fromJson(Map<String, dynamic> json) {
  return GiveawayDetails(
    id: json["id"],
    giveawayLink: json["giveaway_link"] ?? json["post_link"],
    giveawayThumbnail: json["giveaway_thumbnail"],
    winnerImage: json["winner_image"],
    giveawayUniqueCode: json["giveaway_unique_code"] ?? json["campaign_code"],
    giveawayCertificate:
        json["giveaway_certificate"] ?? json['campaign_certificate'],
    clientUserId: json["client_user_id"],
    conductedAt: DateTime.tryParse(json["conducted_at"] ?? ""),
    platform: json["platform"],
    giveawayType: json["giveaway_type"],
    giveawayName: json["giveaway_name"],
    winnerCount: json["winner_count"],
    backupWinnerCount: json["backup_winner_count"],
    giveawayRating: json["giveaway_rating"],
    reviewMessage: json["review_message"],
    involvementType: json["involvement_type"] is int
        ? json["involvement_type"]
        : (json["involvement_type"] == true)
            ? 1
            : 0,
    giveawayStatus: json["giveaway_status"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    deletedAt: json["deleted_at"],
    giveawayPreference: json["giveaway_preference"] == null
        ? null
        : GiveawayPreference.fromJson(json["giveaway_preference"]),
    winners: json["winners"] == null
        ? []
        : List<WinnerDetails>.from(
            json["winners"]!.map((x) => WinnerDetails.fromJson(x))),
  );
}
