factory Giveaway.fromJson(Map<String, dynamic> json) {
  return Giveaway(
    isUnlocked: json['is_unlocked_by_current_user'],
    thumbnail: json['giveaway_thumbnail'] ?? '',
    userName: json['client_user']?['username'],
    id: json["id"],
    giveawayLink: json["giveaway_link"],
    giveawayUniqueCode: json["giveaway_unique_code"],
    giveawayCertificate: json["giveaway_certificate"],
    clientUserId: json["client_user_id"],
    conductedAt: DateTime.tryParse(json["conducted_at"] ?? ""),
    platform: json["platform"],
    giveawayName: json["giveaway_name"],
    giveawayType: json["giveaway_type"],
    winnerCount: json["winner_count"],
    backupWinnerCount: json["backup_winner_count"],
    involvementType: json["involvement_type"],
    giveawayStatus: json["giveaway_status"],
    createdAt: DateTime.tryParse(json["created_at"] ?? ""),
    updatedAt: DateTime.tryParse(json["updated_at"] ?? ""),
    deletedAt: json["deleted_at"],
  );
}
