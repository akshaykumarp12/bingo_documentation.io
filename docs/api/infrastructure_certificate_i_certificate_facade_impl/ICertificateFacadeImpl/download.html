@override
ResultFuture<String> download(DownloadTaskInfo taskInfo) async {
  return runSafely.runSafely(() async {
    await _prepareSaveDir();
    if (localPath == null) {
      throw CustomException(
        errMsg: 'Failed to Save Image',
      );
    }
    final response = await Dio().get(
      taskInfo.link!,
      options: Options(
        responseType: ResponseType.bytes,
        followRedirects: false,
      ),
    );
    if (response.status) {
      final file = File(
          '${localPath!}/Winners_${DateTime.now().millisecondsSinceEpoch}.pdf');
      await file.writeAsBytes(response.data);

      final doc = await PdfDocument.openFile(file.path);
      final page = await doc.getPage(1);
      final pageImage = await page.render(
        width: (page.width * 3).toInt(),
        height: (page.height * 3).toInt(),
      );

      final image = await pageImage.createImageDetached();

      final byteData = await image.toByteData(format: ImageByteFormat.png);
      if (byteData == null) {
        throw CustomException(
          errMsg: 'Failed to Save Image',
        );
      }
      // Save the image to the gallery
      await ImageGallerySaverPlus.saveImage(
        byteData.buffer.asUint8List(),
        quality: 100,
        name: 'Winners_${DateTime.now().millisecondsSinceEpoch}',
      );
      return 'Image Saved';
    } else {
      throw CustomException(
        errMsg: 'Failed to Save Image',
      );
    }
    // } else {
    //   await mediaDownload.downloadMedia(
    //     AppDetails.globalNavigatorKey.currentContext!,
    //     taskInfo.link!,
    //   );
    //   return right('Download started');
    // }
  });
}
