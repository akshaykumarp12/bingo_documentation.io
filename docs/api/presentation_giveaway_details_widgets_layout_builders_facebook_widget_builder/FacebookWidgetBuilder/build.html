@override
Widget build({
  required bool repetedTags,
}) {
  return BlocBuilder<GiveawayDetailsBloc, GiveawayDetailsState>(
    builder: (context, state) {
      return CustomAnimatedTextDropDown(
        isExpand: state.pref == Preferences.rule,
        onTap: () {
          if (state.pref == Preferences.rule) {
            context.read<GiveawayDetailsBloc>().add(
                  const SetVisiblePref(
                    Preferences.none,
                  ),
                );
          } else {
            context.read<GiveawayDetailsBloc>().add(
                  const SetVisiblePref(
                    Preferences.rule,
                  ),
                );
          }
        },
        childHeight: 726,
        decoration: BoxDecoration(
          borderRadius: BorderRadius.circular(16),
          color: ColorResources.secondary,
          boxShadow: [
            BoxShadow(
              offset: const Offset(0, 9),
              blurRadius: 13,
              spreadRadius: 0,
              color: ColorResources.BLACK.withOpacity(0.09),
            )
          ],
        ),
        title: Text(
          'RULES PREFERENCES',
          style: font12W500.textColorLite,
        ),
        child: Column(
          crossAxisAlignment: CrossAxisAlignment.start,
          children: [
            CustomCounter(
              title: 'Total Mentions',
              isOptional: true,
              count: state.totalMentions,
              onDecrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const TotalMentionChanged(-1));
              },
              onIncrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const TotalMentionChanged(1));
              },
            ),
            gapLarge,
            CustomCounter(
              title: 'Total Hashtag',
              isOptional: true,
              count: state.totalHashtags,
              onDecrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const TotalHashtagChanged(-1));
              },
              onIncrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const TotalHashtagChanged(1));
              },
            ),
            gapLarge,
            CustomCounter(
              title: 'Post / Reel Shared',
              isOptional: true,
              count: state.shareCount,
              onDecrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const ShareCountChanged(-1));
              },
              onIncrementClick: () {
                context
                    .read<GiveawayDetailsBloc>()
                    .add(const ShareCountChanged(1));
              },
            ),
            gapLarge,
            CustomSwitchButton(
              title: 'Post/Real Like',
              onChanged: (value) {
                context.read<GiveawayDetailsBloc>().add(
                      NameRequired(value: value),
                    );
              },
              value: state.nameRequired,
            ),
            gapLarge,
            if (repetedTags) ...[
              CustomSwitchButton(
                showHint: true,
                title: 'No Repeated Tags',
                onChanged: (value) {
                  context.read<GiveawayDetailsBloc>().add(
                        NoRepetedTags(value: value),
                      );
                },
                value: state.noRepetedTags,
              ),
              gapLarge
            ],
            RichText(
              text: TextSpan(children: [
                TextSpan(
                  text: 'Pages the user must follow',
                  style: font14W400.w600.textColor,
                ),
                TextSpan(
                  text: ' (Optional)',
                  style: font14W400.s10.textColorLite,
                )
              ]),
            ),
            gapLarge,
            const UserNameWidget(),
            gapLarge,
            GiveawayCriteriaTitle(
              title: 'Comment Likes',
              showHint: false,
              onPressed: () {
                CustomDialog.showCustomDialog(
                  chid: const HintWidget(
                    title: 'Comment Likes',
                    content: 'Total No Of Likes On Comments Of The Post',
                  ),
                );
              },
            ),
            gapLarge,
            CustomTextFormField(
              hintText: 'Total likes',
              fontStyle: FontStyle.italic,
              textInputType: TextInputType.number,
              onChanged: (value) {
                final like = num.tryParse(value);
                if (like == null) return;
                context.read<GiveawayDetailsBloc>().add(
                      TotalCommentLikesChaned(like),
                    );
              },
            ),
            gapLarge,
            GiveawayCriteriaTitle(
              title: 'Keywords',
              isOptional: true,
              showHint: true,
              onPressed: () {
                CustomDialog.showCustomDialog(
                  chid: const HintWidget(
                    title: 'Keyword',
                    content:
                        'If you want your followers to include specific keywords or hashtags in their comments, you can list them here. For example, you might want comments to include terms like “Pencil” or hashtags like “#tbt.” This option will automatically select comments that match the keywords you specify.',
                  ),
                );
              },
            ),
            gapLarge,
            CustomTextFormField(
              hintText: 'Add keyword.....',
              fontStyle: FontStyle.italic,
              onChanged: (value) {
                context.read<GiveawayDetailsBloc>().add(
                      KeyWordChanged(value),
                    );
              },
            ),
            gapLarge,
            if (repetedTags) ...[
              GiveawayCriteriaTitle(
                title: 'Unwanted Words',
                isOptional: true,
                showHint: true,
                onPressed: () {
                  CustomDialog.showCustomDialog(
                    chid: const HintWidget(
                      title: 'Unwanted Words',
                      content:
                          'Enter the words you want to exclude from the giveaway post comments. Users who include these keywords in their comments will be removed from the list.',
                    ),
                  );
                },
              ),
              gapLarge,
              CustomTextFormField(
                hintText: 'Enter unwanted words.....',
                fontStyle: FontStyle.italic,
                onChanged: (value) {
                  context.read<GiveawayDetailsBloc>().add(
                        UnwantedWordChanged(value),
                      );
                },
              ),
            ],
          ],
        ),
      );
    },
  );
}
